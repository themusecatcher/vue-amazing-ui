export function sliceFile(file: File) {
  return new Promise((resolve, reject) => {
    const chunkCount = Math.ceil(file.size / CHUNK_SIZE) // åˆ†ç‰‡æ•°é‡
    const threadChunkCount = Math.ceil(chunkCount / THREAD_COUNT) // æ¯ä¸ªçº¿ç¨‹åˆ†é…çš„åˆ†ç‰‡æ•°é‡

function createFileChunk(file: File, index: number, chunkSize: number) {
  return new Promise((resolve, reject) => {
    const start = index * chunkSize
    const end = Math.min(file.size, start + chunkSize)
